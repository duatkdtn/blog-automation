function fmt(n) {
  return Math.round(n).toLocaleString("ko-KR") + "원";
}

function fmtPreview(val, elId) {
  var el = document.getElementById(elId);
  if (!val || isNaN(val) || parseFloat(val) <= 0) { el.innerHTML = ""; return; }
  var n = parseFloat(val);
  var comma = Math.round(n).toLocaleString("ko-KR");
  var unitStr = "";
  if (n >= 100000000) {
    var uk = Math.floor(n / 100000000);
    unitStr = " (" + uk.toLocaleString("ko-KR") + "억원)";
  } else if (n >= 10000) {
    var man = Math.floor(n / 10000);
    unitStr = " (" + man.toLocaleString("ko-KR") + "만원)";
  } else if (n >= 1000) {
    var chun = Math.floor(n / 1000);
    unitStr = " (" + chun.toLocaleString("ko-KR") + "천원)";
  } else if (n >= 100) {
    var baek = Math.floor(n / 100);
    unitStr = " (" + baek.toLocaleString("ko-KR") + "백원)";
  }
  el.innerHTML = "&#8594; " + comma + "원" + unitStr;
}

function getIncomeTax(monthlyTaxable, dependents) {
  var d = Math.max(1, dependents);
  var tax = 0;
  var m = monthlyTaxable;
  if (m <= 1060000) tax = 0;
  else if (m <= 1500000) tax = Math.round((m - 1060000) * 0.06);
  else if (m <= 3000000) tax = Math.round(26400 + (m - 1500000) * 0.15);
  else if (m <= 4500000) tax = Math.round(251400 + (m - 3000000) * 0.24);
  else if (m <= 8000000) tax = Math.round(611400 + (m - 4500000) * 0.35);
  else if (m <= 15000000) tax = Math.round(1836400 + (m - 8000000) * 0.38);
  else tax = Math.round(4496400 + (m - 15000000) * 0.40);
  var deduction = (d - 1) * 12500;
  return Math.max(0, tax - deduction);
}

function calculate() {
  var annual = parseFloat(document.getElementById("salary").value) || 0;
  var deps = parseInt(document.getElementById("dependents").value) || 1;
  var nontax = parseFloat(document.getElementById("nontax").value) || 0;

  if (annual <= 0) { alert("연봉을 입력해주세요."); return; }

  var monthly = annual / 12;
  var taxable = Math.max(0, monthly - nontax);

  var pension = Math.round(monthly * 0.045);
  var health = Math.round(monthly * 0.03545);
  var care = Math.round(health * 0.1295);
  var employ = Math.round(monthly * 0.009);
  var incomeTax = getIncomeTax(taxable, deps);
  var localTax = Math.round(incomeTax * 0.1);

  var totalDeduction = pension + health + care + employ + incomeTax + localTax;
  var takeHome = Math.round(monthly - totalDeduction);

  document.getElementById("monthly-gross").textContent = fmt(monthly);
  document.getElementById("monthly-deduction").textContent = "-" + fmt(totalDeduction);
  document.getElementById("monthly-take-home").textContent = fmt(takeHome);
  document.getElementById("annual-take-home-sub").textContent = "연간 실수령액: " + fmt(takeHome * 12);
  document.getElementById("pension").textContent = "-" + fmt(pension);
  document.getElementById("health").textContent = "-" + fmt(health);
  document.getElementById("care").textContent = "-" + fmt(care);
  document.getElementById("employ").textContent = "-" + fmt(employ);
  document.getElementById("income-tax").textContent = "-" + fmt(incomeTax);
  document.getElementById("local-tax").textContent = "-" + fmt(localTax);
  document.getElementById("take-home-final").textContent = fmt(takeHome);

  document.getElementById("result-section").style.display = "block";
  document.getElementById("result-section").scrollIntoView({behavior: "smooth"});
}

function setAndCalc(val) {
  document.getElementById("salary").value = val;
  fmtPreview(val, "salary-preview");
  calculate();
}

document.getElementById("salary").addEventListener("input", function() {
  fmtPreview(this.value, "salary-preview");
});
document.getElementById("nontax").addEventListener("input", function() {
  fmtPreview(this.value, "nontax-preview");
});
document.getElementById("salary").addEventListener("keypress", function(e) {
  if (e.key === "Enter") calculate();
});
